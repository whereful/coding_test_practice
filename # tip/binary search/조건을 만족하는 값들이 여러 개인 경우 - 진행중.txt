# 따라서 li의 구간도 정렬되어 있어 binary search 적용 가능

# target 비교 외에 여러 경우 가능
def binary_search(target, start, end):

    # result를 아무 값으로 설정
    result = 0

    while start <= end:
        mid = (start + end) // 2

        # 조건을 만족하지 않고 부족한 경우
        # 최소 지점을 높임
        if li[mid] < target:
            start = mid + 1

        # 조건을 만족하거나 과다한 경우
        # 최대 지점을 낮춤
        elif li[mid] > target:
            end = mid - 1

        # 조건을 만족한 경우
        else:

            start = mid + 1
            
            end = mid - 1

    # 최종 결과를 반환함, 없으면 None 반환
    return result
